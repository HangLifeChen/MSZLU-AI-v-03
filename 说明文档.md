~~~shell
mkdir -p ./etc/systemd/system/containerd.service.d

Environment="HTTPS_PROXY=http://192.168.8.1:7890"
Environment="NO_PROXY=127.0.0.1,localhost,.svc,.cluster.local,10.0.0.0/8,192.168.0.0/16"

~~~


### 3.2 部署NFS

NFS 是一种基于网络的文件存储卷，它允许跨多个节点共享持久化数据。

> 大规模生产环境推荐使用Ceph/Rook

#### 3.2.1 部署NFS服务器

部署在master1（192.168.8.101），有条件的可以部署在一个独立节点上。

~~~shell
sudo apt install nfs-kernel-server
# 切换到root用户
su - 
mkdir -p /data/nfs/mszlu-ai
echo '/data/nfs/mszlu-ai 192.168.8.0/24(rw,sync,no_subtree_check,no_root_squash)' >> ./etc/exports
exportfs -a
systemctl restart nfs-kernel-server
~~~

所有节点都安装nfs客户端：

~~~shell
# 切换到root执行
apt install nfs-common
~~~

部署 nfs-subdir-external-provisioner（master1上安装）：

* 它是一个 **动态供给器**
* 监听 PVC 创建事件
* 在 NFS 服务器上自动创建子目录
* 自动生成 PV
* 把 PV 绑定到 PVC

~~~shell
# 安装helm
curl https://raw.githubusercontent.com/helm/helm/main/scripts/get-helm-3 | bash

helm repo add nfs-subdir-external-provisioner https://kubernetes-sigs.github.io/nfs-subdir-external-provisioner/

helm install nfs-provisioner nfs-subdir-external-provisioner/nfs-subdir-external-provisioner \
      --set nfs.server=192.168.8.101 \
      --set nfs.path=/data/nfs/mszlu-ai \
      --set storageClass.name=nfs-client \
      --set storageClass.defaultClass=true
~~~

## 4. 镜像仓库搭建

~~~shell
# 添加Harbor Helm仓库
helm repo add harbor https://helm.goharbor.io
helm repo update
# 创建命名空间
kubectl create namespace harbor

# 安装Harbor（使用NodePort方式）
helm install harbor harbor/harbor \
  --namespace harbor \
  --create-namespace \
  --set expose.type=nodePort \
  --set expose.tls.enabled=false \
  --set externalURL=http://192.168.8.101:30002 \
  --set harborAdminPassword=Harbor12345 \
  --set persistence.enabled=true \
  --set persistence.persistentVolumeClaim.registry.storageClass=nfs-client \
  --set persistence.persistentVolumeClaim.chartmuseum.storageClass=nfs-client \
  --set persistence.persistentVolumeClaim.jobservice.storageClass=nfs-client \
  --set persistence.persistentVolumeClaim.database.storageClass=nfs-client \
  --set persistence.persistentVolumeClaim.redis.storageClass=nfs-client \
  --set persistence.persistentVolumeClaim.trivy.storageClass=nfs-client

# 查看状态
kubectl get pods -n harbor
kubectl get svc -n harbor
~~~

~~~go
func runBenchmark(config *Config) *Stats {
	stats := &Stats{
		MinDuration: time.Hour, // 初始化为一个很大的值
	}

	// 创建进度控制
	ctx, cancel := make(chan struct{}), func() {}
	closeChan := make(chan struct{})
	go func() {
		time.Sleep(config.Duration)
		close(closeChan)
	}()

	// 创建工作池
	var wg sync.WaitGroup
	resultChan := make(chan RequestResult, config.Concurrency*10)

	// 启动结果收集器
	go collectResults(stats, resultChan)

	// 启动工作协程
	for i := 0; i < config.Concurrency; i++ {
		wg.Add(1)
		go worker(config, &wg, resultChan, closeChan)
	}

	// 启动进度报告
	ticker := time.NewTicker(5 * time.Second)
	go func() {
		for {
			select {
			case <-ticker.C:
				printProgress(stats)
			case <-closeChan:
				ticker.Stop()
				return
			}
		}
	}()

	_ = ctx
	_ = cancel

	// 等待压测完成
	wg.Wait()
	close(resultChan)

	// 等待结果收集完成
	time.Sleep(100 * time.Millisecond)

	return stats
}

func worker(config *Config, wg *sync.WaitGroup, resultChan chan<- RequestResult, stopChan <-chan struct{}) {
	defer wg.Done()

	client := &http.Client{
		Timeout: config.Timeout,
	}

	for {
		select {
		case <-stopChan:
			return
		default:
			result := doRequest(client, config)
			resultChan <- result

			// 检查是否需要停止
			select {
			case <-stopChan:
				return
			default:
			}
		}
	}
}

func doRequest(client *http.Client, config *Config) RequestResult {
	start := time.Now()
	result := RequestResult{}

	// 构建请求体
	reqBody := map[string]interface{}{
		"agentId": config.AgentID,
		"message": config.Message,
	}

	jsonBody, err := json.Marshal(reqBody)
	if err != nil {
		result.Error = fmt.Sprintf("marshal error: %v", err)
		return result
	}

	// 创建请求
	url := config.BaseURL + "/api/v1/agents/chat"
	req, err := http.NewRequest("POST", url, bytes.NewBuffer(jsonBody))
	if err != nil {
		result.Error = fmt.Sprintf("create request error: %v", err)
		return result
	}

	// 设置请求头
	req.Header.Set("Content-Type", "application/json")
	req.Header.Set("Authorization", "Bearer "+config.Token)
	req.Header.Set("Accept", "text/event-stream")

	// 发送请求
	resp, err := client.Do(req)
	if err != nil {
		result.Error = fmt.Sprintf("request error: %v", err)
		return result
	}
	defer resp.Body.Close()

	// 检查状态码
	if resp.StatusCode != http.StatusOK {
		body, _ := io.ReadAll(resp.Body)
		result.Error = fmt.Sprintf("status code: %d, body: %s", resp.StatusCode, string(body))
		return result
	}

	// 读取 SSE 流
	var responseSize int64
	reader := bufio.NewReader(resp.Body)
	for {
		line, err := reader.ReadString('\n')
		if err != nil {
			if err != io.EOF {
				result.Error = fmt.Sprintf("read stream error: %v", err)
				return result
			}
			break
		}
		responseSize += int64(len(line))

		// 检查是否收到 [DONE] 标记
		if bytes.Contains([]byte(line), []byte("[DONE]")) {
			break
		}
	}

	result.Success = true
	result.Duration = time.Since(start)
	result.ResponseSize = responseSize

	return result
}

func collectResults(stats *Stats, resultChan <-chan RequestResult) {
	for result := range resultChan {
		atomic.AddInt64(&stats.TotalRequests, 1)
		atomic.AddInt64(&stats.TotalResponseSize, result.ResponseSize)

		if result.Success {
			atomic.AddInt64(&stats.SuccessRequests, 1)
			stats.mu.Lock()
			stats.TotalDuration += result.Duration
			if result.Duration < stats.MinDuration {
				stats.MinDuration = result.Duration
			}
			if result.Duration > stats.MaxDuration {
				stats.MaxDuration = result.Duration
			}
			stats.mu.Unlock()
		} else {
			atomic.AddInt64(&stats.FailedRequests, 1)
		}
	}
}

func printProgress(stats *Stats) {
	total := atomic.LoadInt64(&stats.TotalRequests)
	success := atomic.LoadInt64(&stats.SuccessRequests)
	failed := atomic.LoadInt64(&stats.FailedRequests)

	fmt.Printf("[进度] 总请求: %d | 成功: %d | 失败: %d | 成功率: %.2f%%\n",
		total, success, failed, float64(success)/float64(total)*100)
}
~~~

## 9. CI/CD 持续集成与部署

### 9.1 CI/CD 概述

CI/CD（持续集成/持续部署）是实现自动化构建、测试和部署的关键实践，能够显著提升开发效率和发布质量。

> 对于 K8s 环境下的 Go 项目，我推荐采用 **GitLab CI + ArgoCD** 的现代化 GitOps 方案。

- **代码提交** → GitLab CI 自动触发构建
- **镜像构建** → 推送到 Harbor 镜像仓库
- **自动部署** → ArgoCD 监听变更并同步到 K8s
- **版本回滚** → 通过 Git 历史快速回滚

#### 9.3.3 创建 GitOps 仓库结构

**GitOps 仓库** 是 GitOps 实践的核心组件，目的是将**应用代码**与**部署配置**分离管理。

为什么需要单独的 GitOps 仓库？

1. 职责分离

| 仓库类型                            | 存放内容                      | 修改者   | 触发动作        |
| :---------------------------------- | :---------------------------- | :------- | :-------------- |
| **应用仓库** (`mszlu-ai`)           | Go 源码、Dockerfile、业务代码 | 开发人员 | CI 构建镜像     |
| **GitOps 仓库** (`mszlu-ai-gitops`) | K8s YAML、Deployment、Service | CI/运维  | ArgoCD 自动部署 |

2. 声明式部署

GitOps 仓库里的 YAML 文件就是集群的**期望状态**：

```yaml
# GitOps 仓库中的 deployment-patch.yaml
spec:
  replicas: 3                    # 期望3个副本
  template:
    spec:
      containers:
      - name: app
        image: mszlu-ai/app:v1.2.3   # 期望的镜像版本
```

ArgoCD 会持续对比**仓库中的配置**与**集群实际状态**，发现不一致就自动同步。

3. 版本化管理部署

```txt
GitOps 仓库历史：
commit A: image: app:v1.0.0   ← 生产运行稳定
commit B: image: app:v1.1.0   ← 发现 bug
commit C: image: app:v1.0.0   ← 快速回滚到 A
```

发现问题直接 `git revert`，ArgoCD 自动回滚。

4. 权限与审计

   - 开发人员：只能推送代码，不能直接改生产配置

   - CI 系统：构建镜像后自动更新 GitOps 仓库

   - 运维人员：通过审查 GitOps 仓库 PR 来控制发布

所有变更都有 Git 记录，**谁、何时、改了什么**一目了然。

5. 多环境管理

```txt
mszlu-ai-gitops/
├── base/                 # 基础配置（共用）
├── overlays/
│   ├── dev/             # 开发环境：1副本、小内存
│   ├── test/            # 测试环境：2副本
│   └── prod/            # 生产环境：3副本、大内存
```

同一份基础配置，不同环境有不同覆盖（Kustomize）。

### 9.4 版本管理与回滚策略

#### 9.4.1 镜像版本管理

推荐采用语义化版本 + Git Commit 的混合标签策略：

| 标签格式                   | 示例                        | 用途               |
| -------------------------- | --------------------------- | ------------------ |
| `latest`                   | `mszlu-ai/app:latest`       | 开发环境，始终最新 |
| `v{major}.{minor}.{patch}` | `mszlu-ai/app:v1.2.3`       | 正式版本发布       |
| `{commit-sha}-{build}`     | `mszlu-ai/app:a1b2c3d-45`   | 每次构建的唯一标识 |
| `{branch}-{build}`         | `mszlu-ai/app:feature-x-12` | 功能分支测试       |

~~~shell
# 版本打标签脚本
#!/bin/bash
VERSION=$1

# 给当前镜像打版本标签
docker tag 192.168.8.101:30002/mszlu-ai/app:latest 192.168.8.101:30002/mszlu-ai/app:${VERSION}
docker push 192.168.8.101:30002/mszlu-ai/app:${VERSION}

echo "✅ 版本 ${VERSION} 已发布"
~~~

#### 9.4.2 K8s 回滚操作

**查看部署历史：**

~~~shell
# 查看部署历史
kubectl rollout history deployment/mszlu-ai-app -n mszlu-ai

# 输出示例：
~~~

**回滚到上一个版本：**


~~~shell

~~~

#### 9.4.3 创建回滚脚本

在 `scripts/rollback.sh` 创建回滚脚本：

~~~shell
#!/bin/bash

# 回滚脚本
set -e

NAMESPACE="mszlu-ai"
DEPLOYMENT_NAME="mszlu-ai-app"

echo "=== 当前部署状态 ==="
kubectl get deployment ${DEPLOYMENT_NAME} -n ${NAMESPACE}

echo ""
echo "=== 部署历史 ==="
kubectl rollout history deployment/${DEPLOYMENT_NAME} -n ${NAMESPACE}

echo ""
read -p "输入要回滚到的版本号（留空则回滚到上一个版本）: " REVISION

if [ -z "$REVISION" ]; then
    echo "正在回滚到上一个版本..."
    kubectl rollout undo deployment/${DEPLOYMENT_NAME} -n ${NAMESPACE}
else
    echo "正在回滚到版本 ${REVISION}..."
    kubectl rollout undo deployment/${DEPLOYMENT_NAME} -n ${NAMESPACE} --to-revision=${REVISION}
fi

echo ""
echo "等待回滚完成..."
kubectl rollout status deployment/${DEPLOYMENT_NAME} -n ${NAMESPACE}

echo ""
echo "=== 回滚后状态 ==="
kubectl get pods -n ${NAMESPACE} -l app=${DEPLOYMENT_NAME}

echo "✅ 回滚完成！"
~~~

赋予执行权限：

~~~shell
chmod +x scripts/rollback.sh
~~~
